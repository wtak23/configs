def load_connectome_matfile(file_list_only=False,pruned=False):
    """ Load design matrix for the structural connectome saved as .mat file.

    Parameters
    ----------
    file_list_only : bool
        Just return list of file-names (to save time)
    pruned : bool
        Get connectomes pruned at density 15%
        
    Returns
    -------
    X : ndarray of shape (n_samples,n_features)=(n,p)
        Data matrix with rows as data points
    fileList : list of shape (n_samples)
        List containing the name of the original image file. 
        Contains subject's ID, which will be used as merge-keys with the
        spreadsheet info.
    

    Examples
    --------
    >>> X, fileList = tw.data.tob_pnc.load_connectome_matfile()
            
    **History**
    
    - 07/05/2016 -- Created.
    """
    if pruned:
        filename = 'tob_pnc_connectome_PROB_pruned_0705_2016.mat'
    else:
        filename = 'tob_pnc_connectome_PROB_0705_2016.mat'
        
    data_path = os.path.join(datapath_tobpnc,filename)
    
    fileList = loadmat(data_path,variable_names=('fileList'))['fileList']
    fileList = fix_file_list(fileList)
    assert sorted(fileList) == fileList
    
    if file_list_only:
        return fileList
    
    X = loadmat(data_path,variable_names=('connMatrix'))['connMatrix']
    X = tw.conn2design(X) # convert (nROI x nROI x nsubj) to (nsub x nEdges)
    
    return X, fileList


[docs]def load_diffusion_volume_matfile(modality='FA',downsample=True,
                                  file_list_only=False):
    """ Load design matrix for the diffusion volume saved as .mat file.

    Parameters
    ----------
    modality : string
        Type of diffusion measure: ``'FA','TR','RD','AX'``
    downsample : bool
        Downsampled volume or not.
    file_list_only : bool
        Just return list of file-names (to save time)
        
    Returns
    -------
    X : ndarray of shape (n_samples,n_features)=(n,p)
        Data matrix with rows as data points
    fileList : list of shape (n_samples)
        List containing the name of the original image file. 
        Contains subject's ID, which will be used as merge-keys with the
        spreadsheet info.
        
    Examples
    --------
    >>> X, fileList = tw.data.tob_pnc.load_diffusion_volume_matfile()
            
    Notes
    -----
    - 07/06/2016 -- Created.
    """
    date_str='0706_2016'
    if downsample:
        filename = 'tobpnc_dsamp_{}_{}.mat'.format(modality,date_str)
    else:
        filename = 'tobpnc_full_{}_{}.mat'.format(modality,date_str)
        
    data_path = os.path.join(datapath_tobpnc,filename)
    
    fileList = loadmat(data_path,variable_names=('fileList'))['fileList']
    fileList = [str(_file) for _file in fileList]
    assert sorted(fileList) == fileList
    
    if file_list_only:
        return fileList
    
    X = loadmat(data_path,variable_names=('X'))['X']
    
    return X, fileList

    
def load_diffusion_volume(modality='FA',dataset='both',downsample=True):
    """ Load design matrix for the diffusion volume.

    Parameters
    ----------
    modality : string
        ``'FA','TR','AX','RD'``
    dataset : string
        ``'both','pnc','tob'``
    downsample : bool
        Downsampled volume
        
    Returns
    -------
    X : ndarray of shape (n_samples,n_features)=(n,p)
        Data matrix with rows as data points
    df : pandas DataFrame of size (n_samples, n_fields)
        Dataframe containing subject phenotypes
        
    >>> X, df = tw.data.tob_pnc.load_connectome()
    >>> age = df['Age_years'].values
    
    History
    -------
    - 07/06/2016 -- Created 
    """
    X,fileList = load_diffusion_volume_matfile(modality,downsample)
    df = load_datasheet()
    
    def file_list_to_id():
        # select substring containing subject id
        return [_subj.split('_'+modality)[0] for _subj in fileList]
    #%% everything below identical to ``load_connectome``
    subj_id = file_list_to_id()
    
    mask = np.array([subj in df['ID'].tolist() for i,subj in enumerate(subj_id)])
    X = X[mask]
    fileList = [fileList[i] for i,_ in enumerate(mask) if mask[i]]
    subj_id  = [subj_id[i]  for i,_ in enumerate(mask) if mask[i]]

    assert subj_id == df['ID'].tolist()
    
    #--- append ``file_name`` onto second column
    df['file_name'] = fileList
    cols = df.columns.tolist()
    cols.insert(1,cols.pop()) # insert last column at 2nd column
    df = df[cols]
    
    if dataset.lower()=='both':
        pass
    elif dataset.lower() == 'tob':
        mask_tob = df.ID.str.startswith('R').values
        df.iloc[mask_tob]
        X = X[mask_tob]
        df = df.iloc[mask_tob].reset_index(drop=True)
    elif dataset.lower() == 'pnc':
        mask_pnc = ~df.ID.str.startswith('R').values
        df.iloc[mask_pnc]
        X = X[mask_pnc]
        df = df.iloc[mask_pnc].reset_index(drop=True)
    return X,df

def load_bct_measures(density=0.15,log_tx=False,return_dict=False,normalized=False):
    """ Load network measures of interest.
    
    BCT scores saved via ``save_0721_bct_weighted_tobpnc.py`` using
    :func:`bct_network_measures`
    
    Parameters
    ----------
    density : float
        Target edge density upon threhsolding (value between (0,1))
    log_tx : bool
        Log transformation
    return_dict : bool
        Return results as dictionary
    normalized : bool
        Return result from *mean-max-edge* normalized connectomes
        
    Usage
    -----
    >>> X = tw.data.pnc.load_connectome()[0]
    >>> BC,CC,deg,eff,ASS,CPL,EFF,TRA = tw.data.pnc.load_bct_measures(log_tx=True)
    
    Returns
    -------
    The following are **nodal measures** of shape (n_subj,n_nodes)
    
    BC : ndarray of shape=(n_subject,n_nodes)
        Betweenness centrality (global measure)
    CC : ndarray of shape=(n_subject,n_nodes)
        Clustering coefficient
    deg : ndarray of shape=(n_subject,n_nodes)
        Degree
    eff : ndarray of shape=(n_subject,n_nodes)
        Local efficiency
        
    The following are **global measures** of shape (n_subj,)
    
    ASS : 
        assortativity
    CPL :
        characteristic path length
    EFF : 
        global efficiency
    TRA :
        transitivity
        
    
    History
    -------
    - 07/21/2016 -- Created 
    - 07/26/2016 -- Updated with option ``normalized``
    """
    if normalized:
        if log_tx:
            file_name = 'bct_meanMaxNormalized_log_weighted_density{:.1f}.mat'.format(100*density)
        else:
            file_name = 'bct_meanMaxNormalized_weighted_density{:.1f}.mat'.format(100*density)
    else:
        if log_tx:
            file_name = 'bct_logweighted_density{:.1f}.mat'.format(100*density)
        else:
            file_name = 'bct_weighted_density{:.1f}.mat'.format(100*density)
    
    file_path = os.path.join(datapath_tobpnc,'bct',file_name)
    
    in_data = loadmat(file_path)
    
    # -- global measures ---
    # flattening applied since shape=(1,nsubj) when saved via savemat
    ASS = in_data['ASS'].ravel()
    CPL = in_data['CPL'].ravel()
    EFF = in_data['EFF'].ravel()
    TRA = in_data['TRA'].ravel()
    
    # -- local measures ---
    BC = in_data['BC']
    CC = in_data['CC']
    deg = in_data['deg']
    eff = in_data['eff']
    
    if return_dict:
        return dict(BC=BC,CC=CC,deg=deg,eff=eff,ASS=ASS,CPL=CPL,EFF=EFF,TRA=TRA)
    else:
        return BC,CC,deg,eff,ASS,CPL,EFF,TRA
        
def load_176roi(dataset='both'):
    """ Load design matrix for the 176 ROI.
    
    Parameters
    ----------
    dataset : string
        ``'both','pnc','tob'``
        
    Returns
    -------
    df176 : pandas DataFrame of shape (n_samples,n_features)=(n,p)
        Data matrix with rows as data points.
        p = 176*(4 FA/TR/AX/RD) * (2 Mean/Std) = 1408.
    df : pandas DataFrame of size (n_samples, n_fields)
        Dataframe containing subject phenotypes
        
    >>> X, df = tw.data.tob_pnc.load_176roi()
    >>> age = df['Age_years'].values
    
    To use only the FA Mean features:
    
    >>> X,df = tw.data.tob_pnc.load_176roi()
    >>> mask = X.columns.str.contains('NZMean.*FA')
    >>> mask += X.columns.str.contains('NZSigma.*FA')
    >>> X = X.iloc[:,mask].values
    
    History
    -------
    - 07/05/2016 -- Created 
    """
    data_path = os.path.join(datapath_tobpnc,'Results','roistats',
                             'PNC_Tobacco_Merged_roistats.csv')
    df176 = pd.read_csv(data_path,index_col=0)
    df176 = df176.sort_index()
    
    df=tw.data.tob_pnc.load_datasheet()
    
    # remove subjects not contained in the phenotypic spreadsheet
    mask = df176.index.isin(df['ID'])
    df176 = df176[mask]
    
    assert df176.index.tolist() == df['ID'].tolist()
    
    if dataset.lower()=='both':
        pass
    elif dataset.lower() == 'tob':
        mask_tob = df.ID.str.startswith('R').values
        df.iloc[mask_tob]
        df176 = df176.ix[mask_tob,:]
        df = df.iloc[mask_tob].reset_index(drop=True)
    elif dataset.lower() == 'pnc':
        mask_pnc = ~df.ID.str.startswith('R').values
        df.iloc[mask_pnc]
        df176 = df176.ix[mask_pnc,:]
        df = df.iloc[mask_pnc].reset_index(drop=True)
    return df176, df
